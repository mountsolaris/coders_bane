DATUMSDIFFERENZ

<form id="myForm">
  <label for="datetime">Datum und Uhrzeit:</label>
  <input type="datetime-local" id="datetime" name="datetime">
  <button type="button" id="calculateBtn">Abstand berechnen</button>
</form>

<script>
  const calculateButton = document.getElementById("calculateBtn");

  calculateButton.addEventListener("click", () => {
    const datetimeInput = document.getElementById("datetime").value;

    if (!datetimeInput) {
      alert("Bitte ein Datum und eine Uhrzeit eingeben!");
      return;
    }

    const inputDate = new Date(datetimeInput);
    const now = new Date();

    const diffMilliseconds = inputDate - now;
    const diffSeconds = Math.floor(diffMilliseconds / 1000);
    const diffMinutes = Math.floor(diffSeconds / 60);
    const diffHours = Math.floor(diffMinutes / 60);
    const diffDays = Math.floor(diffHours / 24);

    alert(`Abstand:
    - Millisekunden: ${diffMilliseconds}
    - Sekunden: ${diffSeconds}
    - Minuten: ${diffMinutes}
    - Stunden: ${diffHours}
    - Tage: ${diffDays}`);
  });
</script>

// Pfad zur TXT-Datei
const txtFilePath = 'example.txt';

// Element, in das der Inhalt eingefügt werden soll
const textContainer = document.getElementById('text-container');

// Lade die TXT-Datei
fetch(txtFilePath)
    .then(response => {
        if (!response.ok) {
            throw new Error(`HTTP-Fehler! Status: ${response.status}`);
        }
        return response.text(); // Inhalt der Datei als Text
    })
    .then(text => {
        // Füge den Text in das HTML-Element ein
        textContainer.innerHTML = text;
    })
    .catch(error => {
        console.error('Fehler beim Laden der Datei:', error);
    });
